安装

npm install --save-dev gulp-coffee

配置gulp

var gulp = require("gulp");
var gutil = require("gulp-util");
var coffee = require('gulp-coffee');

gulp.task('default', function() {
  gulp.src('./coffee/*.coffee')
    .pipe(coffee({bare: true}).on('error', gutil.log))
    .pipe(gulp.dest('./js/'));
});
